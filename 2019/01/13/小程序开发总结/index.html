    function isBlank(str) {

if (Object.prototype.toString.call(str) === '[object Undefined]') {//空

return true

} else if (Object.prototype.toString.call(str) === '[object String]' ||

Object.prototype.toString.call(str) === '[object Array]') { 
字条串或数组

return str.length == 0 ? true : false

} else if (Object.prototype.toString.call(str) === '[object Object]') {

return JSON.stringify(str) == '{}' ? true : false

} else {

return true

}

}
  1. data中数组的赋值
  • 若为数组变量用[]括起来即可
  • 若为数组内的值用“”引起来

  • 若为普通变量可直接赋值

     data: {
indexGoods: [{
           name: null,
           price: null,
           stock: null,
   subphotourl: null
 }],
 iurl : null,
 },

 this.setData({
  [name] : app.globalData.indexdata[i].name,
   [price] : app.globalData.indexdata[i].price,
   [stock] : app.globalData.indexdata[i].stock,
   [subphotourl] : "data:image/jpg;base64," + base64,
   iurl : "data:image/png;base64," + base64,
         });

注意这里的this只能用在第一层函数内 如果要在第二层用 最好提前把this赋值给临时变量
若涉及回调函数也会导致this指针域变化 最好提前赋值给临时变量 或者把success：function（res){}改为success：res=>{}
